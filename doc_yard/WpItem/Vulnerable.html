def vulnerabilities
  xml             = xml(vulns_file)
  vulnerabilities = Vulnerabilities.new

  xml.xpath(vulns_xpath).each do |node|
    vulnerabilities << Vulnerability.load_from_xml_node(node)
  end
  vulnerabilities
end
